Sintaxe

Exemplos de código

JavaScripto usa palavras-chave em português. O restante da sintaxe é idêntico ao JavaScript.

JavaScripto 
funcao saudacao(nome) {
  retorne "Olá, " + nome
}

deixe resultado = saudacao("mundo")
imprima(resultado)

se (resultado.tamanho > 5) {
  imprima("mensagem longa")
} senao {
  imprima("mensagem curta")
}
JavaScript gerado 
function saudacao(nome) {
  return "Olá, " + nome
}

let resultado = saudacao("mundo")
console.log(resultado)

if (resultado.length > 5) {
  console.log("mensagem longa")
} else {
  console.log("mensagem curta")
}
